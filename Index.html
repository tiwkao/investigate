function handleProcess() { const file = document.getElementById('csvInput').files[0]; const name = document.getElementById('fileName').value; if (!file) return alert('กรุณาเลือกไฟล์!'); if (!lineUser) return alert('กรุณารอโหลดโปรไฟล์ LINE สักครู่'); document.getElementById('loading').classList.remove('hidden'); document.getElementById('downloadArea').classList.add('hidden'); document.getElementById('btnMain').disabled = true; const reader = new FileReader(); reader.onload = (e) => { const csvContent = e.target.result; // 1. ส่งข้อมูลไปบันทึก Log ที่ Google Sheets (แบบไม่รอผลตอบกลับเพื่อเลี่ยง CORS) const webAppUrl = 'https://script.google.com/macros/s/AKfybzzEbpnnyZoBzdRxaAniUtGqVSyXDZDYacEedtjnHLXJE5nBdI1r5nxhD9/exec'; fetch(webAppUrl, { method: 'POST', mode: 'no-cors', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ csv: "Log Only", // ไม่ส่งไฟล์ใหญ่ไปทาง Fetch เพื่อความเร็ว name: name, user: lineUser }) }); // 2. ประมวลผลแปลงไฟล์ทันทีบน Browser (ไม่ต้องพึ่ง Google Script) processCsvLocally(csvContent, name); }; reader.readAsText(file); } function processCsvLocally(csvData, name) { try { // แยกบรรทัดและคอลัมน์ (รองรับทั้ง ; และ ,) const rows = csvData.split(/\r?\n/).map(row => { return row.includes(';') ? row.split(';') : row.split(','); }); if (rows.length < 2) throw new Error('ไฟล์ไม่มีข้อมูล'); const headers = rows[0].map(h => h.trim().toUpperCase()); const idx = { plmn: headers.indexOf('PLMN'), tac: headers.indexOf('LAC/TAC'), xci: headers.indexOf('XCI'), enb: headers.indexOf('XNBID'), cid: headers.indexOf('LOCAL_CID'), band: headers.indexOf('BAND'), lat: headers.indexOf('LAT'), lon: headers.indexOf('LON') }; if (idx.lat === -1 || idx.lon === -1) throw new Error('ไม่พบคอลัมน์ LAT หรือ LON'); let result = [["TAC-ECI", "Enb", "PLMN", "BAND", "LATITUDE", "LONGITUDE"]]; let count = 0; for (let i = 1; i < rows.length; i++) { let row = rows[i]; if (row.length > 1 && row[idx.lat] && row[idx.lat] !== "0") { result.push([ `${(row[idx.tac]||"").trim()}-${(row[idx.xci]||"").trim()}`, `${(row[idx.enb]||"").trim()}-${(row[idx.cid]||"").trim()}`, (row[idx.plmn]||"").trim(), (row[idx.band]||"").trim(), row[idx.lat].trim(), row[idx.lon].trim() ]); count++; } } // เก็บผลลัพธ์และแสดงปุ่มดาวน์โหลด convertedRows = result; document.getElementById('loading').classList.add('hidden'); document.getElementById('btnMain').disabled = false; document.getElementById('downloadArea').classList.remove('hidden'); alert('บันทึก Log และประมวลผลสำเร็จ ' + count + ' รายการ'); } catch (err) { document.getElementById('loading').classList.add('hidden'); document.getElementById('btnMain').disabled = false; alert('เกิดข้อผิดพลาด: ' + err.message); } }
